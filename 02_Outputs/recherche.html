# Transformation de la carte de France en grille d'hexagones
France_hex <- France_93 %>%
  st_make_grid(what = "polygons", square = FALSE, n = c(50,25)) %>% 
  st_sf() %>% 
  mutate(id_hex = 1:n()) %>% 
  select(id_hex, geometry)
